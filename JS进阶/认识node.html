认识node

node就是js在服务器中的运行环境（简单理解）
遵循common.js规范
全局变量global
导出require
导入exports或者module.exports

node中的模块分为三类

  1. 自定义模块 自定义文件

  2. 第三方模块 通过npm安装的模块，如：swiper等插件、vue、vuex

  3. 内置模块 下载node时自带的模块，如：路由、HTTP等

上述三个模块引入时有所不同
自定义模块，需要写相对路径
其他两个模块，直接写模块名
三方或内置，先去本地的node_modules中查找；本地若无，则去上级的node_modules中查找；一直查找到根路径的node_modules；若还没有，报错。

npm init 初始化项目，简写npm init -y,使用默认内容

安装jQuery npm i jquery --save
安装less npm i less --sva-dev 上线时不需要的依赖，放在开发环境中

node核心模块—fs 文件系统

readFile 读取文件模块(异步)

异步操作
三个参数 URL，options，callback
URL 读取文件地址
option 编码格式 一般是 utf-8
callback 是回调函数,有两个参数，第一个参数是失败信息，第二个是读取的信息

let fs = require('fs')



//readFile的返回值是undefined；是异步操作

let w=fs.readFile('./test.txt','utf-8',function(err,data){

    if(!err){

        //做一些读取成功的操作

    }

})

console.log(w)//undefined

readFileSync 读取文件模块(同步)

只有两个参数URL，option
读取的数据存在返回值中
读取失败的话，直接报错，不再向下执行

let a = fs.readFileSync('./1.html', 'utf-8')

console.log(a)

writeFile 写文件模块(异步操作)

四个参数：URL、data、option、callback
data是要写入的数据，是字符串格式
默认覆盖原来数据

fs.writeFile('./test.txt','hello 小可爱','utf-8',function(err){



})

writeFileSync 写文件模块(同步操作)

返回值是undefined
如果路径不存在，node会为你新建文件并写入

fs.writeFileSync('./test.txt','hello 小可爱','utf-8')

node核心模块—HTTP

let http = require('http')

let fs = require('fs')

let serve = http.createServer(function (req, res) {

    //req 请求信息

    //req.url      请求路径

    //req.headers  请求头

    //req.method   请求方法



    //res 响应信息

    // res.end('123')

    res.setHeader('qqq','hello')

    if (req.url == '/') {

        let str = fs.readFileSync('./test.html', 'utf-8')

        res.end(str)

    }

    if (req.url == '/favicon.ico') {

        let pic = fs.readFileSync('./1.png')

        res.end(pic)

    }

})

serve.listen('8888', function () {

    //服务启动成功后，会触发该函数

    console.log('服务起在8888端口')

})

express

let express = require("express");

let app = express();

//中间件 次数不限制

app.use(function(req,res,next){

    req.url+='?q=12'

    res.setHeader('qqq','hello')

    next()

})

app.use(function(req,res,next){

    req.url+='?q=12'

    res.setHeader('qqq','hello')

    res.statusCode=500

    res.statusMessage='error'

    next()

})

app.use(express.static('src'))

app.get('/',function (req,res) {

    res.send('qwer')

})

app.listen('8088',function (params) {

    console.log("acjdavn")

})

let http = require('http')

let fs = require('fs')

let url=require('url')

http.createServer(function(req,res){

    //setHeader(响应头的属性，对应的属性值)

    res.setHeader('Access-Control-Allow-Origin','*')



    let {pathname,query}=url.parse(req.url,true)

    console.log(query)

    switch(req.url){

        case '/list':

        fs.readFile('./1.json','utf-8',function(err,data){

            if(!err){

                res.end(data)

            }

        })

        break;

    }

}).listen('9123',function(){

    console.log('服务起于9123')

})
