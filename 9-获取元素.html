获取元素

document.body

document.documentElement

document.getElementById('')//通过ID获取

document.getElementsByTagName('')//通过标签名获取

document.getElementsByClassName('')//通过类名获取   

document.querySelector('放选择器') //只能获取一个

document.querySelectorAll('') //获取全部

节点

一个页面只有一个根节点（document）

四大节点

  • 元素节点
    • nodeType:1
    • nodeName:大写标签名
    • nodeValue:null
  • 文本节点
    • nodeType:3
    • nodeName:#text
    • nodeValue:文本内容
  • 注释节点
    • nodeType:8
    • nodeName:#commet
    • nodeValue:注释内容
  • 根节点
    • nodeType:9
    • nodeName:#document
    • nodeValue:null

兄弟/子元素获取

var h1=document.querySelector('h1');

h1.previousSibling;//哥哥节点

h1.previousElementSibling;//哥哥元素节点，不兼容IE8



h1.nextSibling;//弟弟节点

h1.nextElementSibling;//弟弟元素节点，不兼容IE8





h1.childNodes;//儿子节点

h1.children;//儿子元素节点



h1.parentNode;//父亲节点



var ele=document.createElement();//创造元素

ele.innerHTML='哈哈哈';

h2.appendChild(ele);//给他塞个儿子
