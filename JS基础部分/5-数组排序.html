冒泡排序 双for



var ary=[10,2,1,3,4,2,1,2]

for (var j=0;j<ary.length;j++){



for (var i=0;i<ary.length-1;i++){

    var cur=ary[i]

    if (ary[i]>ary[i+1]){

        ary[i]=ary[i+1]

        ary[i+1]=cur

    }

}

}

快速排序 递归法

function fn(ary) {

    //边界

    if(ary.length<=1){

        return ary;

    }



    var cur=ary.splice(parseInt(ary.length/2))[0]

    var right=[]

    var left=[]

    for (var i=0;i<ary.length;i++){

        if(cur>ary[i]){

            left.push(ary[i])

        }else{

            right.push(ary[i])

        }

    }

    return fn(left).concat(cur,fn(right))

}
