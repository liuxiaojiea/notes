小动画的不同实现方式

固定步长

    var box = document.getElementById('box');

    let maxl = utilObj.win('clientWidth') - utilObj.css(box, 'width')

    var timer = setInterval(function () {

        let l = utilObj.css(box, 'left');

        if (l >= maxl) {

            clearInterval(timer);

            utilObj.css(box, 'left', maxl);

            return

        }

        utilObj.css(box, 'left', ++l)

    }, 2)

固定时间

    let s = 800;//路程时800

    let t = 3000//3000毫秒走完

    let speed = s / t;

    let time = 0;//存储运动的总时间 

    let timer = setInterval(() => {

        time += 20;



        if (time >= t) {

            time = t;

            clearInterval(timer)

        }

        //time总时间*spend 速度为总路程

        let l = time*speed;

        //让盒子的left值变成l即可;

        box.style.left=l+"px";





    }, 20)

回到顶部

    var btn = document.querySelector('.btn')



    //固定时间

    btn.flag = true;

    btn.onclick = function () {

        if (!this.flag) return;

        this.flag = false;

        let t = utilObj.win('scrollTop');

        let speed = t / 1000;

        var time = 0;

        var timer = setInterval(function () {

            time += 50;

            if (time >= 1000) {

                time = 1000;

                clearInterval(timer);

            }

            utilObj.win('scrollTop', t - time * speed)

        }, 50)

    }
