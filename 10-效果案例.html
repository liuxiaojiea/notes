随机生成验证码

//得到n位不同的数

function getAry(n) {

    n = n||4;

    var ary4=[];

    while(ary4.length<n){

        var m=Math.round(Math.random()*61);

        if(ary4.indexOf(m)==-1){

            ary4.push(m);

        }

    }

    return ary4;

}

    //3、获取时间差

    var time = target-now;

   if(time<=0){
